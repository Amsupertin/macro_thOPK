automacro autoChangeClass {
	priority 0 
	exclusive 1
	run-once 1
	location izlude,izlude_a,izlude_b,izlude_c,izlude_d,izlude_e,izlude_f,izlude_g,izlude_h,prt_fild08,prt_fild08a,prt_fild08b,prt_fild08c,prt_fild08d,prt_fild08e,prt_fild08f,prt_fild08g,prt_fild08h
	job == 10
	class Novice
	call {
		do conf route_randomWalk 0
		$classNow = @config(classNow)
		$classChange = @config(classChange)
		
		#Thief
		if (($.map == izlude || $.map == izlude_a || $.map == izlude_b || $.map == izlude_c || $.map == izlude_d || $.map == izlude_e || $.map == izlude_f || $.map == izlude_g || $.map == izlude_h ) && $classNow == Novice && $classChange == Thief) goto autoChangeClassThief1
		if (($.map == prt_fild08 || $.map == prt_fild08a || $.map == prt_fild08b || $.map == prt_fild08c || $.map == prt_fild08d || $.map == prt_fild08e || $.map == prt_fild08f || $.map == prt_fild08g || $.map == prt_fild08h )  && $classNow == Novice && $classChange == Thief) goto autoChangeClassThief2

		#Acolyte
		if (($.map == izlude || $.map == izlude_a || $.map == izlude_b || $.map == izlude_c || $.map == izlude_d || $.map == izlude_e || $.map == izlude_f || $.map == izlude_g || $.map == izlude_h )  && $classNow == Novice && $classChange == Acolyte) goto autoChangeClassAcolyte1
		if (($.map == prt_fild08 || $.map == prt_fild08a || $.map == prt_fild08b || $.map == prt_fild08c || $.map == prt_fild08d || $.map == prt_fild08e || $.map == prt_fild08f || $.map == prt_fild08g || $.map == prt_fild08h )  && $classNow == Novice && $classChange == Acolyte) goto autoChangeClassAcolyte2
		
		:autoChangeClassThief1
		do move iz_ac01
		do move $.map 78 25
		goto Thief
		
		:autoChangeClassThief2
		do move izlude
		do move iz_ac01
		do move $.map 78 25	
		goto Thief
		
		:autoChangeClassAcolyte1
		do move iz_ac01
		do move $.map 78 25
		goto Acolyte
		
		:autoChangeClassAcolyte2
		do move izlude
		do move iz_ac01
		do move $.map 78 25	
		goto Acolyte
		
		:Thief
		do move $.map 46 140
		pause 5
		do talknpc 46 142 c c r0 c r1 c n
		do conf classNow Thief
		do conf macroStep onLeveling
		do move izlude
		call getWarpTicket
		stop
		
		:Acolyte
		do move $.map 153 169
		pause 5
		do talknpc 156 173 c c r0 c r1 c n
		do conf classNow Acolyte
		do conf macroStep onLeveling
		do move izlude
		call getWarpTicket
		stop
	}
}

macro getWarpTicket {
	do move 52 172
	do move 69 176
	do talknpc 69 177 c r2 c r0 c r0 n
	do talknpc 69 177 c r1 r0 n
	do move 74 158
}

automacro autoAdd {
	priority 0
	exclusive 1
	timeout 300
	call {
		#Thief
		if ($classNow == Thief && $classChange == Thief) goto autoAddThief
		#Acolyte
		if ($classNow == Acolyte && $classChange == Acolyte) goto autoAddAcolyte
	
		:autoAddThief
		do conf statsAddAuto 1
		do conf statsAddAuto_list  10 dex, 15 agi, 20 dex, 25 agi, 20 str, 30 dex, 60 agi, 30 str, 40 dex, 20 vit, 80 agi, 25 luk ,60 dex, 50 str, 40 luk, 90 agi, 60 str, 29 vit
		do conf statsAddAuto_dontUseBonus 1
		do conf statsAdd_over_99 1
		[
		log ==========================================
		log = StatsAddAuto [Thief] Successful
		log ==========================================
		]
		do conf skillsAddAuto 1
		do conf skillsAddAuto_list TF_DOUBLE 10, TF_MISS 10, TF_STEAL 10, TF_HIDING 8, TF_POISON 10, TF_DETOXIFY 1
		[
		log ==========================================
		log = SkillAddAuto [Thief] Successful
		log ==========================================
		]
		stop
		
		:autoAddAcolyte
		do conf statsAddAuto 1
		do conf statsAddAuto_list 40 int, 40 dex, 70 int, 70 dex
		do conf statsAddAuto_dontUseBonus 1
		do conf statsAdd_over_99 1
		[
		log ==========================================
		log = StatsAddAuto [Acolyte] Successful 
		log ==========================================
		]
		do conf skillsAddAuto 1
		do conf skillsAddAuto_list AL_HEAL 3, AL_INCAGI 10, AL_HEAL 10, AL_DP 5, AL_BLESSING 10, AL_CURE 1, AL_RUWACH 1, AL_TELEPORT 2, AL_WARP 4, AL_HOLYWATER 1, AL_ANGELUS 10
		[
		log ==========================================
		log = SkillAddAuto [Acolyte] Successful
		log ==========================================
		]
		stop
	}
}
